CREATE TABLE GYMS (
	G_ID varchar(40) PRIMARY KEY,
	G_LAT DECIMAL(2,6),
	G_LON DECIMAL(2,6),
    G_NAME varchar(300),
	G_IMG varchar(300)
);

CREATE TABLE GYM_DATA (
	GD_ID INTEGER PRIMARY KEY AUTOINCREMENT,
	GD_ID_GYM INTEGER,
	GD_TIMESTAMP INTEGER,
	GD_POINTS INTEGER,
	GD_LEVEL INTEGER,
	GD_OWNER_TEAM INTEGER,
    GD_IS_IN_BATTLE BOOLEAN,
	FOREIGN KEY (GD_ID_GYM) REFERENCES GYMS(G_ID),
	FOREIGN KEY (GD_LEVEL) REFERENCES LEVELS(L_ID),
	FOREIGN KEY (GD_OWNER_TEAM) REFERENCES TEAMS(T_ID)
);
